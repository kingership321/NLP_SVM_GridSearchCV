def processing(txt):
    preprocess_txt = []
    doc = nlp(txt)
    for token in doc:
        if token.is_stop or token.is_punct:
            continue
        preprocess_txt.append(token.lemma_)
    return " ".join(preprocess_txt)

print(doc)
